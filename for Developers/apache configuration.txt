<virtualhost *:80>
	ServerName *
	ServerAlias *
	DocumentRoot /var/www/

	RewriteEngine on

	RewriteCond %{HTTP_HOST} ^([a-zA-Z0-9\-]+).([a-z]+)$ [NC]
	RewriteRule (.*) http://www.%{HTTP_HOST}/$1 [QSA,R=301] 

	RewriteCond %{REQUEST_URI} ^/(.*)/$
	RewriteRule (.*) %1 [L,R=301]

	RewriteCond %{HTTP_HOST} ^([a-zA-Z0-9\-]+).([a-zA-Z0-9\-]+).([a-z]+)$ [NC]
	RewriteRule (.*) %{DOCUMENT_ROOT}/%3/%2/%1/$1 [L,QSA]
</virtualhost>
